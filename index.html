    <p id="result"></p>


    <script>

        async function searchDomain() {

            const domain = document.getElementById('domain').value;

            const resultElement = document.getElementById('result');


            if (!domain) {

                resultElement.textContent = 'Please enter a domain name.';

                return;

            }


            // Replace with your actual Domainr API key

            const apiKey = 'YOUR_DOMAINR_API_KEY';

            const apiUrl = `https://api.domainr.com/v2/search?mashape-key=${apiKey}&query=${domain}`;


            try {

                const response = await fetch(apiUrl);

                const data = await response.json();


                if (data.results) {

                    const count = data.results.length;

                    resultElement.textContent = `"${domain}" has been registered in ${count} TLDs.`;

                } else {

                    resultElement.textContent = `Error fetching data for "${domain}".`;

                }

            } catch (error) {

                resultElement.textContent = 'Error fetching data.';

            }

        }

    </script>
